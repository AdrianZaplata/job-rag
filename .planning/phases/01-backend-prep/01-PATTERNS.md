```python
import asyncio
from datetime import datetime, timezone
from fastapi import Request
from sse_starlette.sse import EventSourceResponse, ServerSentEvent
from job_rag.agent.stream import stream_agent
from job_rag.api.sse import AgentEvent, ErrorEvent, HeartbeatEvent, to_sse
from job_rag.config import settings

def _heartbeat_factory() -> ServerSentEvent:
    ev = HeartbeatEvent(type="heartbeat", ts=datetime.now(timezone.utc).isoformat())
    return ServerSentEvent(event="heartbeat", data=ev.model_dump_json())

def _sanitize(exc: BaseException) -> str:
    return str(exc).strip().replace("\n", " ")[:200] or "internal error"

@router.get("/agent/stream", dependencies=[Depends(require_api_key), Depends(agent_limit)])
async def agent_stream(request: Request, q: str) -> EventSourceResponse:
    app = request.app

    async def typed_event_generator():
        current_task = asyncio.current_task()
        if current_task is not None:
            app.state.active_streams.add(current_task)
        try:
            try:
                async with asyncio.timeout(settings.agent_timeout_seconds):
                    async for event in stream_agent(q):
                        yield to_sse(event)
            except asyncio.TimeoutError:
                yield to_sse(ErrorEvent(
                    type="error", reason="agent_timeout",
                    message=f"Agent exceeded {settings.agent_timeout_seconds}s timeout",
                ))
            except asyncio.CancelledError:
                yield to_sse(ErrorEvent(
                    type="error", reason="shutdown",
                    message="Server is shutting down — please retry shortly",
                ))
                raise
            except Exception as e:
                yield to_sse(ErrorEvent(
                    type="error", reason="internal", message=_sanitize(e),
                ))
        finally:
            if current_task is not None:
                app.state.active_streams.discard(current_task)

    return EventSourceResponse(
        typed_event_generator(),
        ping=settings.heartbeat_interval_seconds,
        ping_message_factory=_heartbeat_factory,
        shutdown_event=app.state.shutdown_event,
        shutdown_grace_period=30.0,
        headers={"X-Accel-Buffering": "no", "Content-Encoding": "identity"},
    )
```

**Deviation notes:**
- `asyncio.timeout` (3.11+ context manager) is used instead of `asyncio.wait_for` — cleaner inside the generator. D-25 allows either; wire-effect identical.
- The sanitization helper `_sanitize` is defensive per D-19 — never leaks stack traces.
- Re-raising `CancelledError` in the `except` branch is required so the task actually terminates after the error event is emitted (standard asyncio cancellation discipline).
- `stream_agent(q)` signature unchanged; `event` now comes back as a Pydantic `AgentEvent` (see File 4) — `to_sse` converts to sse-starlette dict.
- Separately, inject `get_current_user_id` in `/match/{posting_id}` (line 82) and `/gaps` (line 100) per D-10 — signature change is `user_id: UUID = Depends(get_current_user_id)` even though body passes it to `load_profile(user_id=user_id)` which doesn't branch on it yet.

from datetime import datetime
from typing import Annotated, Literal, Union
from pydantic import BaseModel, Field

class TokenEvent(BaseModel):
    type: Literal["token"]
    content: str   # wire-shape parity with current stream.py line 42

class ToolStartEvent(BaseModel):
    type: Literal["tool_start"]
    name: str
    args: dict | None = None

class ToolEndEvent(BaseModel):
    type: Literal["tool_end"]
    name: str
    output: str

class HeartbeatEvent(BaseModel):
    type: Literal["heartbeat"]
    ts: str    # ISO-8601

class ErrorEvent(BaseModel):
    type: Literal["error"]
    reason: Literal["agent_timeout", "shutdown", "llm_error", "internal"]
    message: str   # sanitized — NEVER a stack trace

class FinalEvent(BaseModel):
    type: Literal["final"]
    content: str

AgentEvent = Annotated[
    Union[TokenEvent, ToolStartEvent, ToolEndEvent, HeartbeatEvent, ErrorEvent, FinalEvent],
    Field(discriminator="type"),
]

    raw_text = file_path.read_text(encoding="utf-8")

DEFAULT_UUID_PATTERN = re.compile(
    r"DEFAULT.*['\"]?[0-9a-f-]{36}['\"]?.*::?uuid|"
    r"server_default\s*=.*[Uu][Uu][Ii][Dd]",

    """CI guard: no Alembic migration may add a DDL DEFAULT to a user_id column.

### Pydantic v2 BaseModel with discriminated enum

**Source analog:** `src/job_rag/models.py` lines 40-46, lines 1-38 for enum style.
**Apply to:** `src/job_rag/api/sse.py`.

**Pattern:**
- `StrEnum` for closed string sets (used elsewhere in the codebase: `RemotePolicy`, `Seniority`).
- For SSE events, use `Literal[...]` on the `type` field (not StrEnum) — the discriminated-union mechanism in Pydantic v2 requires the `type` to be a Literal match, not an enum instance.
- `Field(description="...")` on all fields for OpenAPI clarity.

---

### SQLAlchemy 2.x `Mapped[]` ORM columns

**Source analog:** `src/job_rag/db/models.py` lines 11-46 (`JobPostingDB`).
**Apply to:** New `UserDB`, `UserProfileDB`.

**Pattern:**
- `Mapped[type] = mapped_column(...)` — not the older `Column(...)` syntax.
- Primary keys with `uuid.UUID` and `primary_key=True`.
- `server_default=func.now()` for timestamps; `DateTime(timezone=True)` for timezone awareness on new tables.
- `__table_args__` tuple for indexes (not decorators).
- **Critical deviation for user_id**: NO `default=uuid.uuid4` AND NO `server_default` (D-08, Pitfall 18).
