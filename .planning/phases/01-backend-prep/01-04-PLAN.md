```python
"""SSE event contract for /agent/stream.

Pydantic v2 discriminated union on `type` field per D-14. Six event types:
- token: assistant text chunk
- tool_start: tool invocation begun (tool name + args)
- tool_end: tool invocation completed (tool name + output)
- heartbeat: 15s liveness ping — emitted by sse-starlette ping_message_factory
- error: timeout / shutdown / llm error / internal — emitted by route handler
- final: stream complete marker

Wire-shape parity: the JSON these models emit MUST be byte-identical to the
current agent/stream.py dicts. The frontend (Phase 6) already consumes this
shape; changing it is a breaking contract change.
"""
from datetime import datetime  # noqa — allowed for future ts use
from typing import Annotated, Literal, Union

from pydantic import BaseModel, Field


class TokenEvent(BaseModel):
    type: Literal["token"]
    content: str = Field(description="Assistant text chunk")


class ToolStartEvent(BaseModel):
    type: Literal["tool_start"]
    name: str = Field(description="Tool function name")
    args: dict | None = Field(default=None, description="Tool arguments (JSON)")


class ToolEndEvent(BaseModel):
    type: Literal["tool_end"]
    name: str = Field(description="Tool function name")
    output: str = Field(description="Tool return value (stringified)")


class HeartbeatEvent(BaseModel):
    type: Literal["heartbeat"]
    ts: str = Field(description="ISO-8601 timestamp at emit time")


class ErrorEvent(BaseModel):
    type: Literal["error"]
    reason: Literal["agent_timeout", "shutdown", "llm_error", "internal"] = Field(
        description="Machine-readable error kind"
    )
    message: str = Field(description="Sanitized human-readable message — NEVER a stack trace")


class FinalEvent(BaseModel):
    type: Literal["final"]
    content: str = Field(description="Synthesized final assistant message")


AgentEvent = Annotated[
    Union[TokenEvent, ToolStartEvent, ToolEndEvent, HeartbeatEvent, ErrorEvent, FinalEvent],
    Field(discriminator="type"),
]


def to_sse(event: BaseModel) -> dict[str, str]:
    """Convert a Pydantic event to the sse-starlette ServerSentEvent payload dict."""
    return {
        "event": event.type,  # type: ignore[attr-defined]
        "data": event.model_dump_json(),
    }
```
