## 4. DashboardFilters Component Anatomy

**Location:** `frontend/src/components/dashboard/DashboardFilters.tsx`

**Purpose:** Single horizontal flex row of three controls — country `DropdownMenu`, seniority `DropdownMenu`, remote `ToggleGroup`. Reads + writes filter state via `useDashboardFilters()`.

### JSX skeleton (locked)

```tsx
// frontend/src/components/dashboard/DashboardFilters.tsx
import { ChevronDown } from 'lucide-react'

import { Button } from '@/components/ui/button'
import {
  DropdownMenu,
  DropdownMenuCheckboxItem,
  DropdownMenuContent,
  DropdownMenuRadioGroup,
  DropdownMenuRadioItem,
  DropdownMenuTrigger,
} from '@/components/ui/dropdown-menu'
import { ToggleGroup, ToggleGroupItem } from '@/components/ui/toggle-group'

import { useDashboardFilters } from './useDashboardFilters'

const COUNTRY_LABEL: Record<'PL' | 'DE' | 'EU' | 'WW', string> = {
  PL: 'Poland',
  DE: 'Germany',
  EU: 'EU',
  WW: 'Worldwide',
}

const SENIORITY_LABEL: Record<'junior' | 'mid' | 'senior' | 'staff' | 'lead', string> = {
  junior: 'Junior',
  mid: 'Mid',
  senior: 'Senior',
  staff: 'Staff',
  lead: 'Lead',
}

export function DashboardFilters() {
  const { filters, setFilters } = useDashboardFilters()

  return (
    <div
      className="flex flex-col gap-2 md:flex-row md:items-center"
      role="group"
      aria-label="Dashboard filters"
    >
      {/* Country */}
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button variant="outline" size="sm" className="justify-between md:w-40">
            {COUNTRY_LABEL[filters.country]}
            <ChevronDown className="h-4 w-4 opacity-50" aria-hidden="true" />
          </Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent align="start">
          <DropdownMenuRadioGroup
            value={filters.country}
            onValueChange={(v) => setFilters({ country: v as typeof filters.country })}
          >
            <DropdownMenuRadioItem value="WW">Worldwide</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="EU">EU</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="DE">Germany</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="PL">Poland</DropdownMenuRadioItem>
          </DropdownMenuRadioGroup>
        </DropdownMenuContent>
      </DropdownMenu>

      {/* Seniority */}
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button variant="outline" size="sm" className="justify-between md:w-40">
            {filters.seniority ? SENIORITY_LABEL[filters.seniority] : 'Any seniority'}
            <ChevronDown className="h-4 w-4 opacity-50" aria-hidden="true" />
          </Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent align="start">
          <DropdownMenuRadioGroup
            value={filters.seniority ?? ''}
            onValueChange={(v) =>
              setFilters({ seniority: v === '' ? undefined : (v as typeof filters.seniority) })
            }
          >
            <DropdownMenuRadioItem value="">Any seniority</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="junior">Junior</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="mid">Mid</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="senior">Senior</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="staff">Staff</DropdownMenuRadioItem>
            <DropdownMenuRadioItem value="lead">Lead</DropdownMenuRadioItem>
          </DropdownMenuRadioGroup>
        </DropdownMenuContent>
      </DropdownMenu>

      {/* Remote */}
      <ToggleGroup
        type="single"
        value={filters.remote}
        onValueChange={(v) => {
          if (v) setFilters({ remote: v as typeof filters.remote })
        }}
        aria-label="Remote policy"
        size="sm"
      >
        <ToggleGroupItem value="any" aria-label="Any remote policy">
          Any
        </ToggleGroupItem>
        <ToggleGroupItem value="remote" aria-label="Remote only">
          Remote
        </ToggleGroupItem>
        <ToggleGroupItem value="non_remote" aria-label="On-site or hybrid only">
          On-site
        </ToggleGroupItem>
      </ToggleGroup>
    </div>
  )
}
```

### Control contract

| Control | shadcn primitive | Values | Default | URL elided when default? |
|---------|------------------|--------|---------|-------------------------|
| Country | `DropdownMenu` + `DropdownMenuRadioGroup` | `PL` \| `DE` \| `EU` \| `WW` | `WW` | YES (`country=WW` omitted from URL) |
| Seniority | `DropdownMenu` + `DropdownMenuRadioGroup` | `''` \| `junior` \| `mid` \| `senior` \| `staff` \| `lead` | `''` (undefined) | YES (omitted when undefined) |
| Remote | `ToggleGroup type="single"` + 3 `ToggleGroupItem` | `any` \| `remote` \| `non_remote` | `any` | YES (`remote=any` omitted from URL) |

### Control labels (verbatim)

- Country trigger when selected: `Worldwide` / `EU` / `Germany` / `Poland`
- Country items (in dropdown order): `Worldwide`, `EU`, `Germany`, `Poland`
- Seniority trigger when none selected: `Any seniority`
- Seniority items: `Any seniority`, `Junior`, `Mid`, `Senior`, `Staff`, `Lead`
- Remote toggle items: `Any`, `Remote`, `On-site`
- Filter bar group label: `aria-label="Dashboard filters"`
- Remote group label: `aria-label="Remote policy"`

### Default-elision behaviour (D-17)

When `setFilters({...})` is called:
- If the new value equals the default (`WW` for country, `undefined`/`''` for seniority, `any` for remote), call `params.delete(key)` instead of `params.set(key, value)`.
- This keeps URLs clean: `/dashboard` (no params) === `/dashboard?country=WW&remote=any`.

### Disabled states

- **None disabled.** Per CONTEXT D-19: "filter bar always interactive" — even when a widget is loading or erroring, all three controls remain enabled. Changing a filter while a widget is loading cancels the in-flight TanStack Query and starts a fresh fetch (TanStack Query default behaviour with abortable `signal`).

### Keyboard contract

- Tab order: country trigger → seniority trigger → remote toggle (first item) → (within toggle group, arrow keys move between items per Radix `ToggleGroup` defaults).
- Country / seniority dropdowns: arrow keys navigate items, `Enter`/`Space` selects, `Escape` closes.
- Remote toggle: arrow keys navigate, `Space` selects (Radix default).
- All triggers get `focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2` (inherited from shadcn primitives).

### Bar row anatomy

Each skill row inside `<SkillsBarList>`:

```tsx
function SkillRow({ skill, must, nice, total, max }: {
  skill: string
  must: number
  nice: number
  total: number
  max: number  // = skills[0].total (the leader); bar widths are proportional to leader
}) {
  const widthPct = Math.round((total / max) * 100)
  const mustPct = Math.round((must / total) * 100)
  return (
    <div className="grid grid-cols-[1fr_auto] items-center gap-2 text-sm">
      <div className="min-w-0">
        <div className="flex items-center justify-between gap-2">
          <span className="truncate font-mono text-xs">{skill}</span>
          <span className="text-xs text-muted-foreground tabular-nums">
            {total} ({must}m / {nice}n)
          </span>
        </div>
        {/* Two-segment bar: solid must portion + outlined nice portion */}
        <div
          className="mt-1 h-1.5 w-full overflow-hidden rounded-sm bg-muted"
          role="presentation"
        >
          <div
            className="h-full bg-foreground"
            style={{ width: `${widthPct}%` }}
            aria-hidden="true"
          >
            <div
              className="h-full bg-primary"
              style={{ width: `${mustPct}%` }}
              aria-hidden="true"
            />
          </div>
        </div>
      </div>
    </div>
  )
}
```

### Width formula

Bar width = `(skill.total / leader.total) * 100%`. The first (highest-count) skill always renders at 100% width; every other bar is scaled relative to the leader. This is "leader-relative" not "max-of-corpus-relative" — keeps bars readable at any filter set.

Inside each bar, the must / nice split is visualized via two stacked segments:
- Outer bar (full row width): `bg-foreground` (subtle, indicates total)
- Inner bar (must-have only): `bg-primary` (accent, indicates must-have count); width = `(must / total) * 100%` of the outer bar.

### Numeric label format

`{total} ({must}m / {nice}n)` — example: `64 (52m / 12n)`. Right-aligned, `tabular-nums`, `text-xs text-muted-foreground`. Bar visualization is decorative (`aria-hidden`); the numeric label is the accessible source of truth.

## 10. URL State Contract

The dashboard syncs filter state to `?country=&seniority=&remote=` URL search params. Deep-linkable, refresh-safe (REQUIREMENTS DASH-06).

### Param table (locked)

| Param | Values | Default | URL elided when default? | Notes |
|-------|--------|---------|--------------------------|-------|
| `country` | `PL` \| `DE` \| `EU` \| `WW` | `WW` | YES | When omitted → `WW`. Invalid values (e.g. `?country=ZZ`) are coerced to `WW` (defensive) — UI treats unknown values as default. |
| `seniority` | `junior` \| `mid` \| `senior` \| `staff` \| `lead` | `undefined` (no filter) | YES (when undefined) | Invalid values → undefined (defensive). `unknown` is technically a valid backend value but NOT surfaced in the UI per CONTEXT D-08. |
| `remote` | `any` \| `remote` \| `non_remote` | `any` | YES | Invalid values → `any`. |

### Set semantics

`setFilters(patch)` calls `setSearchParams((prev) => next)` with a fresh `URLSearchParams`. Each key in the patch:

1. If `patch[key] === DEFAULT[key]` (or undefined for seniority), call `next.delete(key)`.
2. Otherwise call `next.set(key, patch[key])`.

This ensures URLs stay clean — `/dashboard` (no params) is canonical for the all-defaults state.

### Deep-link table (locked examples)

| URL | Resulting filter state | Widget behaviour |
|-----|------------------------|------------------|
| `/dashboard` | `{ country: 'WW', seniority: undefined, remote: 'any' }` | All 3 widgets fetch unfiltered (all postings) |
| `/dashboard?country=DE` | `{ country: 'DE', seniority: undefined, remote: 'any' }` | 3 widgets fetch with `country=DE` |
| `/dashboard?country=DE&seniority=senior&remote=remote` | `{ country: 'DE', seniority: 'senior', remote: 'remote' }` | 3 widgets fetch with all 3 filters |
| `/dashboard?country=WW&seniority=mid` | `{ country: 'WW', seniority: 'mid', remote: 'any' }` | After first render, URL is rewritten to `/dashboard?seniority=mid` (default elision applies on next `setFilters` call; immediate URL is NOT mutated on mount — see "Mount semantics" below) |

### Mount semantics

On `Dashboard.tsx` mount:
- `useSearchParams` reads the current URL as-is.
- `useDashboardFilters()` returns the parsed filters with defaults applied.
- The URL is NOT rewritten on mount. A URL like `/dashboard?country=WW` stays as-is until the user touches a filter.
- **Rationale:** mutating the URL on mount would cause a routing side-effect during render — anti-pattern. Default elision applies only on user-driven `setFilters` calls.

### Refresh / back-button behaviour

- Refresh: URL preserved → filters preserved → React Query refetches (or hits cache if within `staleTime`).
- Browser back/forward: `useSearchParams` reacts to history change → filters update → React Query refetches with new key.

### `useDashboardFilters` hook contract (typed)

**Location:** `frontend/src/components/dashboard/useDashboardFilters.ts`

```ts
import { useSearchParams } from 'react-router'
import type { Seniority } from '@/api/types'  // openapi-typescript output

export type Country = 'PL' | 'DE' | 'EU' | 'WW'
export type Remote = 'any' | 'remote' | 'non_remote'

export type DashboardFilters = {
  country: Country
  seniority: Seniority | undefined
  remote: Remote
}

const DEFAULT_COUNTRY: Country = 'WW'
const DEFAULT_REMOTE: Remote = 'any'
const COUNTRIES: readonly Country[] = ['PL', 'DE', 'EU', 'WW'] as const
const REMOTES: readonly Remote[] = ['any', 'remote', 'non_remote'] as const

function isCountry(v: string | null): v is Country {
  return v !== null && (COUNTRIES as readonly string[]).includes(v)
}

function isRemote(v: string | null): v is Remote {
  return v !== null && (REMOTES as readonly string[]).includes(v)
}

export function useDashboardFilters() {
  const [params, setParams] = useSearchParams()

  const countryRaw = params.get('country')
  const remoteRaw = params.get('remote')
  const seniorityRaw = params.get('seniority')

  const filters: DashboardFilters = {
    country: isCountry(countryRaw) ? countryRaw : DEFAULT_COUNTRY,
    seniority: (seniorityRaw as Seniority | null) ?? undefined,
    remote: isRemote(remoteRaw) ? remoteRaw : DEFAULT_REMOTE,
  }

  function setFilters(patch: Partial<DashboardFilters>) {
    setParams(
      (prev) => {
        const next = new URLSearchParams(prev)
        if ('country' in patch) {
          if (patch.country && patch.country !== DEFAULT_COUNTRY) {
            next.set('country', patch.country)
          } else {
            next.delete('country')
          }
        }
        if ('seniority' in patch) {
          if (patch.seniority) {
            next.set('seniority', patch.seniority)
          } else {
            next.delete('seniority')
          }
        }
        if ('remote' in patch) {
          if (patch.remote && patch.remote !== DEFAULT_REMOTE) {
            next.set('remote', patch.remote)
          } else {
            next.delete('remote')
          }
        }
        return next
      },
      { replace: false },
    )
  }

  return { filters, setFilters }
}
```

Note: `replace: false` (push new history entry) — allows browser back-button to step through filter combinations.
