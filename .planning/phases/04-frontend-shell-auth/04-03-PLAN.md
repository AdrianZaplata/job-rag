            run: uv python install 3.12

          - name: Install backend dependencies (for codegen-drift-check)
            run: uv sync --frozen

          - name: Start FastAPI in background
            env:
              DATABASE_URL: postgresql://postgres:postgres@localhost:5432/test
              ASYNC_DATABASE_URL: postgresql+asyncpg://postgres:postgres@localhost:5432/test
              ALLOWED_ORIGINS: http://localhost:5173
              SEEDED_USER_ENTRA_OID: ""
              ENTRA_TENANT_ID: "00000000-0000-0000-0000-000000000000"
              ENTRA_TENANT_SUBDOMAIN: "jobrag"
              BACKEND_AUDIENCE: "api://00000000-0000-0000-0000-000000000000"
            run: |
              uv run alembic upgrade head
              uv run uvicorn job_rag.api.app:app --host 127.0.0.1 --port 8000 &
              for i in {1..30}; do
                curl -fsS http://127.0.0.1:8000/openapi.json >/dev/null 2>&1 && break
                sleep 1
              done
              curl -fsS http://127.0.0.1:8000/openapi.json > /tmp/openapi.live.json
              # Sanity: live OpenAPI must parse and contain components.
              python -c "import json; d=json.load(open('/tmp/openapi.live.json')); assert 'components' in d, 'OpenAPI missing components'"

          - name: Codegen drift-check (frontend/openapi.snapshot.json vs live)
            run: |
              # Compare ignoring formatting differences via canonical JSON dump.
              python -c "
              import json, sys
              live = json.load(open('/tmp/openapi.live.json'))
              snap = json.load(open('frontend/openapi.snapshot.json'))
              if json.dumps(live, sort_keys=True) != json.dumps(snap, sort_keys=True):
                  print('::error::OpenAPI snapshot drift — update frontend/openapi.snapshot.json AND re-run codegen')
                  sys.exit(1)
              print('OpenAPI snapshot matches live')
              "
