**File 5: `scripts/refresh-swa-origin.sh`** — DEPL-12 helper, executable, idempotent per A3 + RESEARCH.md lines 643-660:

```bash
#!/bin/bash
# DEPL-12 / Plan-Locking Addendum A3: Two-Pass CORS Bootstrap helper.
# Reads the SWA default origin from terraform output, rewrites prod.tfvars,
# and re-runs `terraform apply -var-file=prod.tfvars`.
#
# Idempotent: safe to re-run. If `swa_origin` already matches the current
# SWA default host, the second `apply` is a no-op revision.
#
# This script is intentionally one-shot infra glue (per the user's reusable-
# tools default — not folded into the `job-rag` Typer CLI; reuse expected
# near-zero).

set -euo pipefail
cd "$(dirname "$0")/../infra/envs/prod"

# Read SWA default host name from terraform state (must come from a TF output)
SWA_ORIGIN="https://$(terraform output -raw swa_default_origin)"

# Re-write prod.tfvars in place
if grep -q '^swa_origin' prod.tfvars; then
  sed -i.bak "s|^swa_origin.*|swa_origin = \"$SWA_ORIGIN\"|" prod.tfvars
else
  echo "swa_origin = \"$SWA_ORIGIN\"" >> prod.tfvars
fi

terraform apply -var-file=prod.tfvars -auto-approve
```

After writing, ensure executable bit: `chmod +x scripts/refresh-swa-origin.sh` (PATTERNS.md line 369; matches `Dockerfile` line 37 convention `RUN chmod +x /app/docker-entrypoint.sh`).

**File 6: `.github/workflows/static-tf.yml`** — static TF validation workflow. Mirrors `.github/workflows/ci.yml` structure but runs only on PRs touching `infra/**`:

```yaml
name: Static TF

on:
  pull_request:
    branches: [main, master]
    paths:
      - 'infra/**'
      - '.github/workflows/static-tf.yml'

jobs:
  static-tf:
    runs-on: ubuntu-latest
    permissions:
      contents: read
    steps:
      - uses: actions/checkout@v4

      - uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: 1.9.5
          terraform_wrapper: false

      - name: Terraform fmt
        run: terraform fmt -check -recursive infra/

      - name: Setup tflint
        uses: terraform-linters/setup-tflint@v4
        with:
          tflint_version: latest

      - name: Init tflint
        run: tflint --init
        working-directory: infra
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Run tflint
        run: tflint --recursive --config="$(pwd)/.tflint.hcl"
        working-directory: infra

      - name: Setup tfsec
        uses: aquasecurity/tfsec-action@v1.0.3
        with:
          working_directory: infra
          additional_args: --config-file infra/.tfsec/config.yml

      # Terraform validate runs PER env directory (validate doesn't recurse usefully)
      - name: Terraform validate (envs/prod)
        run: |
          if [ -f infra/envs/prod/main.tf ]; then
            cd infra/envs/prod && terraform init -backend=false && terraform validate
          fi

      - name: Terraform validate (envs/dev)
        run: |
          if [ -f infra/envs/dev/main.tf ]; then
            cd infra/envs/dev && terraform init -backend=false && terraform validate
          fi

      - name: Terraform validate (bootstrap)
        run: |
          if [ -f infra/bootstrap/main.tf ]; then
            cd infra/bootstrap && terraform init -backend=false && terraform validate
          fi
```
