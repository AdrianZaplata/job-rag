```yaml
  push:
    branches: [master]

  workflow_dispatch:

permissions:

jobs:
  apply:
    name: terraform apply (prod)
    runs-on: ubuntu-latest
    # Protected environment — Adrian as the sole required reviewer.
    # Federated credential matches subject "repo:<owner>/<repo>:environment:production"
    # (Plan 04 identity module's azuread_application_federated_identity_credential.production_env).
    environment: production
    defaults:
      run:
        working-directory: infra/envs/prod
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Terraform
        uses: hashicorp/setup-terraform@v3
        with:
          terraform_version: 1.9.5
          terraform_wrapper: false

      - name: Azure login (OIDC)
        uses: azure/login@v2
        with:
          client-id:       ${{ secrets.AZURE_CLIENT_ID }}        # GHA app reg client_id (workforce tenant per A4)
          tenant-id:       ${{ secrets.AZURE_TENANT_ID }}        # workforce tenant ID
          subscription-id: ${{ secrets.AZURE_SUBSCRIPTION_ID }}

      - name: Terraform Init
        run: terraform init -input=false

      - name: Terraform Apply
        run: terraform apply -input=false -auto-approve -var-file=prod.tfvars
        # Sensitive var values come from environment-scoped GitHub Actions secrets via TF_VAR_* env vars.
        env:
          TF_VAR_ghcr_pat:             ${{ secrets.GHCR_PAT }}
          TF_VAR_openai_api_key:       ${{ secrets.OPENAI_API_KEY }}
          TF_VAR_langfuse_public_key:  ${{ secrets.LANGFUSE_PUBLIC_KEY }}
          TF_VAR_langfuse_secret_key:  ${{ secrets.LANGFUSE_SECRET_KEY }}
```

```yaml
name: Deploy API

on:
  push:
    branches: [master]
    paths:
      - 'src/**'

      - 'uv.lock'
      - 'Dockerfile'
      - 'alembic/**'
      - 'scripts/docker-entrypoint.sh'
      - '.github/workflows/deploy-api.yml'

permissions:
  id-token: write   # OIDC handshake to Azure
  contents: read
  packages: write   # GHCR push

jobs:
  build-and-deploy:
    name: build → push GHCR → ACA revision
    runs-on: ubuntu-latest
    # NO environment gate — push to master is the federated credential subject
    # (Plan 04 identity module's azuread_application_federated_identity_credential.master).
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      # B3: After first push, the GHCR package's visibility must be set to public
      # (recommended for portfolio repo per A2) OR var.ghcr_pat must have read:packages
      # scope on the package. See infra/envs/prod/README.md "Image push: GHCR visibility"
      # for the runbook (one-time portal click after first `docker push`).
      - name: GHCR login
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Setup buildx
        uses: docker/setup-buildx-action@v3

      - name: Build and push image
        uses: docker/build-push-action@v6
        with:
          context: .
          push: true
          tags: |
            ghcr.io/${{ github.repository }}:${{ github.sha }}
            ghcr.io/${{ github.repository }}:latest
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Azure login (OIDC)
        uses: azure/login@v2
        with:
          client-id:       ${{ secrets.AZURE_CLIENT_ID }}
          tenant-id:       ${{ secrets.AZURE_TENANT_ID }}
          subscription-id: ${{ secrets.AZURE_SUBSCRIPTION_ID }}

      - name: Update Container App image
        run: |
          az containerapp update \
            --name jobrag-prod-api \
            --resource-group jobrag-prod-rg \
            --image "ghcr.io/${{ github.repository }}:${{ github.sha }}"

      - name: Smoke /health after revision swap
        run: |
          # Wait up to 90s for the new revision to be active and reachable.
          ACA_FQDN="$(az containerapp show \
            --name jobrag-prod-api \
            --resource-group jobrag-prod-rg \
            --query properties.configuration.ingress.fqdn -o tsv)"

          for i in {1..18}; do
            if curl -fs --max-time 5 "https://${ACA_FQDN}/health" >/dev/null; then
              echo "Health check passed after $((i*5))s"
              exit 0
            fi
            sleep 5
          done
          echo "Health check did not pass after 90s — revision may be unhealthy"
          exit 1
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner ← OIDC token from `id-token: write` permission | Short-lived per-job token; matches federated credential subject; RG-scoped Contributor caps blast radius. |
| GitHub Actions runner ← `secrets.AZURE_STATIC_WEB_APPS_API_TOKEN_PROD` | Long-lived token (180 days); rotates via deploy-infra.yml's post-apply step. |
| GitHub Actions runner ← `secrets.GITHUB_TOKEN` | Built-in per-job token; scope limited by `permissions:` block. |
| Workforce tenant SP → Azure subscription | RG-scoped Contributor only. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-02 (HIGH) | Spoofing | OIDC trust over-privileged | mitigate | All three workflows reference the workforce tenant SP via `secrets.AZURE_*`; RG-scoped Contributor (Plan 04 identity module) caps blast radius to `jobrag-prod-rg`. NO PR trigger. NO workflow_dispatch on deploy-api.yml or deploy-spa.yml (only on deploy-infra.yml, gated by `environment: production`). |
| T-3-04 (MEDIUM) | Information Disclosure | GHCR push token | mitigate | `secrets.GITHUB_TOKEN` is per-job + scoped via `permissions: { packages: write }`. The runtime PAT (`var.ghcr_pat`) is separate — used by ACA at pull time, NOT by the workflow at push time. |
| T-3-05 (MEDIUM) | Information Disclosure | SWA api_key rotation | mitigate | **B2: SWA api_key sync is a MANUAL runbook step** in `infra/envs/prod/README.md` ("Phase-close: GitHub secrets sync").
